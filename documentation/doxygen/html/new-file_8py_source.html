23def build_docs():
24 print("--> NeuraBSD: Generating documentation...")

26 os.chdir(BASE_DIR)

32def serve_docs():
33 try:
34 if not os.path.exists(HTML_DIR):
35 print(f"ERROR: {HTML_DIR} does not exist.")
36 sys.exit(1)

38 os.chdir(HTML_DIR)
39 handler = http.server.SimpleHTTPRequestHandler
40 socketserver.TCPServer.allow_reuse_address = True
41
42 with socketserver.TCPServer(("", PORT), handler) as httpd:
43 url = f"http://localhost:{PORT}"
44 print(f"--> NeuraBSD: Preview available at {url}")
45 print("--> Press Ctrl+C to stop the server.")
46
47 # Iridium Start
48 subprocess.Popen(["iridium", url],
49 stdout=subprocess.DEVNULL,
50 stderr=subprocess.DEVNULL)

52 httpd.serve_forever()
53 except KeyboardInterrupt:
54 print("\n--> NeuraBSD: Documentation server stopped.")
